require('dialogue')

local d = Dialogue:new()

d:seq{
  start=true,
  name="test",
  {text="Welcome to Caminho-Unity!"},
  {text="If you're reading this then the dialogue loaded successfully."},
  {text="Good bye!"}
}

ch={{text="Yes", next="yesSelected"}, 
    {text="No", next="noSelected"}}

d:add{name="choice",
      package="dialogue2", 
      text="Is a hot dog a sandwich?",
      choices=ch, 
      start=true}
      
d:add{package="dialogue2", name="yesSelected", text="I'm not sure I agree..."}
d:add{package="dialogue2", name="noSelected", text="I don't think so either!"}

return d
